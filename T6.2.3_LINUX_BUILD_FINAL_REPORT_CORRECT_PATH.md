   docker run --rm \
     -v "$(pwd)":/home/rust/src \
     -w /home/rust/src \
     --user "$(id -u)":"$(id -g)" \
     -e TAURI_PRIVATE_KEY="" \
     -e TAURI_KEY_PASSWORD="" \

     cargo tauri build --target x86_64-unknown-linux-gnu
