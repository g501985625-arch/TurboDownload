   ```bash
   cd ~/.openclaw/workspace/projects/TurboDownload
   docker run --rm \
     -v "$(pwd)":/home/rust/src \
     -w /home/rust/src \
     --user "$(id -u)":"$(id -g)" \
     -e TAURI_PRIVATE_KEY="" \
     -e TAURI_KEY_PASSWORD="" \
     ghcr.io/tauri-apps/tauri:debian \
     cargo tauri build --target x86_64-unknown-linux-gnu
   ```
